Procedimiento Resultado_Apuesta (E: seleccion, multiplicador, número, E/S: apuesta, saldo_disponible, fichas_ganadas, fichas_perdidas, S: Partidas_Ganadas_Modo1, Partidas_Ganadas_Modo2, Partidas_Perdidas_Modo1, Partidas_Perdidas_Modo2);
Inicio
  Mostrar << "Ok, veamos que tocó";	
  Si (Número = Seleccion) Entonces,
  		Mostrar << "Maravilloso, has ganado";
  		saldo_disponible = saldo_disponible + (apuesta * multiplicador);
  		fichas_ganadas = fichas_ganadas + (apuesta * multiplicador);
  		Si (modo_elegido = 1) Entonces;
        	Partidas_Ganadas_Modo1 = Partidas_Ganadas_Modo1 + 1;
  		Fin_Si
  		Si (modo_elegido = 2) Entonces;
        	Partidas_Ganadas_Modo2 = Partidas_Ganadas_Modo2 + 1;
  		Fin_Si
  Sino
  	Mostrar << "Lamentablemente has perdido";
  	saldo_disponible = saldo_disponible - (apuesta * 1,5);
	  fichas_perdidas = fichas_perdidas + (apuesta * 1,5);
    Si (modo_elegido = 1) Entonces;
      Partidas_Perdidas_Modo1 = Partidas_Perdidas_Modo1 + 1;
  	Fin_Si
  	Si (modo_elegido = 2) Entonces;
      Partidas_Perdidas_Modo2 = Partidas_Perdidas_Modo2 + 1;
    Fin_Si
  Fin_Si
Fin_Procedimiento
