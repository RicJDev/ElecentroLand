Procedimiento Restaurante(E/S: opcion_restaurante, i_restaurante, restaurante_ingredientes_cantidades[i_restaurante], restaurante_contadores[i_restaurante]: Entero, restaurante_ingredientes_precios[i_restaurante]: Real);
Inicio
  Var R = 0: Entero;
  Repetir
    Bienvenida(opcion_restaurante);
    Mostrar << Bienvenida;

    En_Caso (opcion_restaurante > 0) Sea;
      Caso (opcion_restaurante = 1);
        Desayuno(cantidad, cuenta_cliente, monto_pago, dinero_restaurante, i_restaurante, restaurante_ingredientes_cantidades[i_restaurante], restaurante_contadores[i_restaurante], restaurante_ingredientes_precios[i_restaurante], validacion_pago);
        Mostrar << Desayuno;
      Caso (opcion_restaurante = 2);
        Almuerzo(cantidad, cuenta_cliente, monto_pago, dinero_restaurante, i_restaurante, restaurante_ingredientes_cantidades[i_restaurante], restaurante_contadores[i_restaurante], restaurante_ingredientes_precios[i_restaurante], validacion_pago);
        Mostrar << Almuerzo;
      Caso (opcion_restaurante = 3);
        Cena(cantidad, cuenta_cliente, monto_pago, dinero_restaurante, i_restaurante, restaurante_ingredientes_cantidades[i_restaurante], restaurante_contadores[i_restaurante], restaurante_ingredientes_precios[i_restaurante], validacion_pago);
        Mostrar << Cena;
      Otro_Caso
        R = 4;
    Fin_Caso

    Hasta (R = 4);
  Fin_Repetir
Fin_Procedimiento
