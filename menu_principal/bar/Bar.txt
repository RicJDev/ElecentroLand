Procedimiento Bar(E/S: opcion_inv, opcion_clientes: Entero, bar_ingredientes_cantidades[i_bar], bar_contadores[i_bar]: Entero, cuenta_cliente, bar_ingredientes_precios[i_bar], dinero_bar: Real);
Inicio
  Var R: Booleano;

  Repetir
    Mostrar << "¡Bienvenido a nuestro Bar Elecentro Drinks!";
    Mostrar << "¿Qué desea ordenar?";
    Mostrar << "1: Alimentos";
    Mostrar << "2: Bebidas";
    Leer >> opcion_inv;
    Mientras (opcion_inv < 1 or opcion_inv > 2) Hacer;
      Mostrar << "Error en datos, ingrese nuevamente";
      Leer >> opcion_inv;
    Fin_Mientras

    Si (opcion_inv = 1) Entonces;
      Alimentos(opcionclientes, bar_ingredientes_cantidades[i_bar], bar_contadores[i_bar], cuenta_cliente, dinero_bar, bar_ingredientes_precios[i_bar]);
      Mostrar << Alimentos;

    Sino
      Si (opcion_inv = 2) Entonces;
        Bebidas(opcionclientes, bar_ingredientes_cantidades[i_bar], bar_contadores[i_bar], cuenta_cliente, dinero_bar, bar_ingredientes_precios[i_bar]);
        Mostrar << Bebidas;
      Fin_Si
    Fin_Si

    Mostrar << "¿Desea ingresar otro producto? 1: Sí, 0: No";
    Leer >> R;
    Hasta (R = 0);
  Fin_Repetir
Fin_Procedimiento
