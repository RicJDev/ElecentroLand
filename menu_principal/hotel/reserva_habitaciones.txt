Procedimiento reserva_habitaciones(E/S: fecha_salida[i], fecha_llegada[i]: Cadena, nueva_id, dias_hospedaje[i], opcion_habitacion[i], cantidad_habitacion[i]: Entero, total_pagar_habitacion[i], monto_pago, ingresos_hotel: Real, S: validacion_pago: Booleano);
Inicio
  Var cantidad_habitacion: Entero;
  Mostrar << "Seleccione una opción";
  Mostrar << "1: Habitación Elecentro (Habitación individual: Diseñada para una persona, con cama individual) precio 10$ por noche";
  Mostrar << "2: Habitación Guanabana (habitacion con dual) precio: 20$ por noche";
  Mostrar << "3: Habitación Maravillas (hasta 4 personas) precio: 35$ por noche";
  Mostrar << "4: Habitación Picha (hasta 6 personas) precio: 45$ por noche";
  Leer >> opcion_habitacion[nueva_id];

  Mientras (opcion_habitacion[nueva_id] < 1 OR opcion_habitacion[nueva_id] > 4) Hacer;
    Mostrar << "Error por favor elija una habitacion disponible";
    Leer >> opcion_habitacion[nueva_id];
  Fin_Mientras

  En_Caso (opcion_habitacion[nueva_id] > 0) Sea;
    Caso (opcion_habitacion[nueva_id] = 1);

      Mientras (cantidad_habitacion[1] < 1) Hacer;
        Mostrar << "Disculpe la habitacion seleccionada ya esta llena";
        Leer >> opcion_habitacion[nueva_id];
      Fin_Mientras

      cantidad_habitacion = cantidad_habitacion[1];
      cantidad_habitacion = cantidad_habitacion - 1;
      cantidad_habitacion[1] = cantidad_habitacion;

      Mostrar << "¿Cuantos dias desea hospedarse?";
      Leer >> dias_hospedaje[nueva_id];

      Mientras (dias_hospedaje[nueva_id] < 1) Hacer;
        Mostrar << "Por favor ingrese una cantidad valida";
        Leer >> dias_hospedaje[nueva_id];
      Fin_Mientras

      Mostrar << "Señale su fecha de llegada DD-MM-AA";
      Leer >> fecha_llegada[nueva_id];
      Mostrar << "Señale su fecha de salida DD-MM-AA";
      Leer >> fecha_salida[nueva_id];
      total_pagar_habitacion[nueva_id] = 10 * dias_hospedaje[nueva_id];
      ingresos_hotel = ingresos_hotel + total_pagar_habitacion[nueva_id];
      Mostrar << "Estimado cliente usted ha elegido la habitacion Elecentro";

    Caso (opcion_habitacion[nueva_id] = 2);

      Mientras (cantidad_habitacion[2] < 1) Hacer;
        Mostrar << "Disculpe la habitacion seleccionada ya esta llena";
        Leer >> opcion_habitacion[nueva_id];
      Fin_Mientras

      cantidad_habitacion = cantidad_habitacion[2];
      cantidad_habitacion = cantidad_habitacion - 1;
      cantidad_habitacion[2] = cantidad_habitacion;

      Mostrar << "¿Cuantos dias desea hospedarse?";
      Leer >> dias_hospedaje[nueva_id];

      Mientras (dias_hospedaje[nueva_id] < 1) Hacer;
        Mostrar << "Por favor ingrese una cantidad valida";
        Leer >> dias_hospedaje[nueva_id];
      Fin_Mientras

      Mostrar << "Señale su fecha de llegada DD-MM-AA";
      Leer >> fecha_llegada[nueva_id];
      Mostrar << "Señale su fecha de salida DD-MM-AA";
      Leer >> fecha_salida[nueva_id];
      total_pagar_habitacion[nueva_id] = 20 * dias_hospedaje[nueva_id];
      ingresos_hotel = ingresos_hotel + total_pagar_habitacion[nueva_id];

      Mostrar << "Estimado cliente usted ha elegido la habitacion Guanábana";

    Caso (opcion_habitacion[nueva_id] = 3);

      Mientras (cantidad_habitacion[3] < 1) Hacer;
        Mostrar << "Disculpe, la habitación seleccionada ya esta llena";
        Leer >> opcion_habitacion;
      Fin_Mientras

      cantidad_habitacion = cantidad_habitacion[3];
      cantidad_habitacion = cantidad_habitacion - 1;
      cantidad_habitacion[3] = cantidad_habitacion;

      Mostrar << "¿Cuantos dias desea hospedarse?";
      Leer >> dias_hospedaje[nueva_id];

      Mientras (dias_hospedaje[nueva_id] < 1) Hacer;
        Mostrar << "Por favor ingrese una cantidad valida";
        Leer >> dias_hospedaje[nueva_id];
      Fin_Mientras

      Mostrar << "Señale su fecha de llegada DD-MM-AA";
      Leer >> fecha_llegada[nueva_id];
      Mostrar << "Señale su fecha de salida DD-MM-AA";
      Leer >> fecha_salida[nueva_id];
      total_pagar_habitacion[nueva_id] = 35 * dias_hospedaje[nueva_id];
      ingresos_hotel = ingresos_hotel + total_pagar_habitacion[nueva_id];

      Mostrar << "Estimado cliente usted ha elegido la Habitación Maravillas";

    Otro_Caso

      Mientras (cantidad_habitacion[4] < 1) Hacer;
        Mostrar << "Disculpe la habitacion seleccionada ya esta llena";
        Leer >> opcion_habitacion;
      Fin_Mientras

      cantidad_habitacion = cantidad_habitacion[4];
      cantidad_habitacion = cantidad_habitacion - 1;
      cantidad_habitacion[4] = cantidad_habitacion;

      Mostrar << "¿Cuantos dias desea hospedarse?";
      Leer >> dias_hospedaje[nueva_id];

      Mientras (dias_hospedaje[nueva_id] < 1) Hacer;
        Mostrar << "Por favor ingrese una cantidad valida";
        Leer >> dias_hospedaje[nueva_id];
      Fin_Mientras

      Mostrar << "Señale su fecha de llegada DD-MM-AA";
      Leer >> fecha_llegada[nueva_id];
      Mostrar << "Señale su fecha de salida DD-MM-AA";
      Leer >> fecha_salida[nueva_id];
      total_pagar_habitacion[nueva_id] = 45 * dias_hospedaje[nueva_id];
      ingresos_hotel = ingresos_hotel + total_pagar_habitacion[nueva_id];
      Mostrar << "Estimado cliente usted ha elegido la Habitación Picha";
  Fin_Caso
  monto_pago = total_pagar_habitacion[nueva_id];
  metodo_pago(monto_pago, validacion_pago);
  Mostrar << metodo_pago;
  Si (validacion_pago = 1) Entonces;
    Mostrar << "El pago ha sido realizado correctamente, gracias por confiar en nosotros";
  Sino
    Mostrar << "Paga pelabolas.";
  Fin_Si
Fin_Procedimiento
