Procedimiento Almuerzo(E/S: Cantidad: Entero, Cuenta_Cliente: Real, i_restaurante: Entero, restaurante_ingredientes[]: Cadena, restaurante_ingredientes_cantidades[]: Entero, restaurante_contadores[]: Entero);
Inicio
  Var opcion: Entero;
  Var quieres_otra: Booleano;
  Var ingrediente_1, ingrediente_2, ingrediente_3: Entero;

  Var R: Booleano;
  
  Repetir
    Mostrar << "Este es nuestro menú de Almuerzo:";
    Mostrar << "Opción 1: Ensalada Cesar y de bebida Vino Blanco: 32$";
    Mostrar << "Opción 2: Sopa de Cebolla y tomate, y de bebida Vino Tinto: 25$";
    Mostrar << "Opción 3: Bistec de Carne de Res y de bebida Bourbon: 22$";
    Mostrar << "Opción 4: Chuletas de Cerdo a la BBQ y de bebida John Collins: 25$";
    Mostrar << "Opción 5: Pasta Ex´ reprobada y de bebida Estracto de Picha: 19$";
    Mostrar << "Opción 6: Ensalada de Guanábana a la APK y de bebida Whisky Maravilla: 27$";
    Leer >> opcion;
    
    Mientras (opcion < 1) OR (opcion > 6) Hacer;
      Mostrar << "Ha seleccionado una opción fuera del menú, intente nuevamente";
      Leer >> opcion;
    Fin_Mientras
    
    En_Caso (opcion > 0) Sea;
      Caso (opcion = 1);
        ingrediente_1 = restaurante_ingredientes[10];
        ingrediente_2 = restaurante_ingredientes[11];
        
        Repetir
          Si (ingrediente_1 > 0 AND ingrediente_2 > 0) Entonces;
            Mostrar << "Un racion de Ensalada Cesar y una copa de bebida Vino Blanco. Serian, 32$"; // 11 y 10
            
            ingrediente_1 = ingrediente_1 - 1;
            ingrediente_2 = ingrediente_2 - 1;
            
            restaurante_ingredientes[10] = ingrediente_1;
            restaurante_ingredientes[11] = ingrediente_2;
            
            Mostrar << "¿Quieres otro combo igual, manao? (0: No; 1: Si)";
            Leer >> quieres_otra;
            
            Sino 
            
            Mostrar << "Mira, mijo. Se acabo";
            quieres_otra = 0;
          Fin_Si
          
          Hasta (quieres_otra = 0);
        Fin_Repetir
      Caso (opcion = 2);
        Mientras (Cantidad > Cantidad_Cebolla OR Cantidad_Tomate OR Cantidad > Vinotinto) Hacer;
          Mostrar << "Solo hay ", Cantidad_Cebolla, "Cebollas,  ", Cantidad_Tomate, "Tomates y ", Vinotinto, "Botellas de Vinotinto disponibles";
          Leer >> Cantidad;
        Fin_Mientras
        
        Opc_Selec2 = Opc_Selec2 + 1;
        Cuenta_Cliente = Cuenta_Cliente + (Cantidad * Opc2);
        Dinero_Opc2 = Dinero_Opc2 + (Cantidad * Opc2);
        Cantidad_Cebolla = Cantidad_Cebolla – Cantidad;
        Cantidad_Tomate = Cantidad_Tomate – Cantidad;
        Vinotinto = Vinotinto – Cantidad;
        Mostrar << "Usted selecciono la Opción 2 (Sopa de Cebolla y tomate, y de bebida Vino Tinto)  con un precio de 25$";
      Caso (opcion = 3);
        Mostrar << "¿Cuántas ordenó?";
        Leer >> Cantidad;
        
        Mientras (Cantidad > Cantidad_Bistec_Res OR Cantidad > Bourbon) Hacer;
          Mostrar << "Solo hay ", Cantidad_Bistec_Res, "Bistecs de Res y ", Bourbon, "Botellas de Bourbon disponibles";
          Leer >> Cantidad;
        Fin_Mientras
        
        Opc_Selec3 = Opc_Selec 3 + 1;
        Cuenta_Cliente = Cuenta_Cliente + (Cantidad * Opc3);
        Dinero_Opc3 = Dinero_Opc3 + (Cantidad * Opc3);
        Cantidad_Bistec_Res = Cantidad_Bistec_Res – Cantidad;
        Bourbon = Bourbon – Cantidad;
        Mostrar << "Usted selecciono la Opción 3 (Bistec de Carne de Res y de bebida Bourbon) con un precio de 22$";
      Caso (opcion = 4);
        Mostrar << "¿Cuántas ordenó?";
        Leer >> Cantidad;
        
        Mientras (Cantidad > Cantidad_Chuletas OR Cantidad > John_Collins) Hacer;
          Mostrar << "Solo hay ", Cantidad_Chuletas, "Chuletas y ", John_Collins, "Botellas de John Collins disponibles";
          Leer >> Cantidad;
        Fin_Mientras
        
        Opc_Selec4 = Opc_Selec 4 + 1;
        Cuenta_Cliente = Cuenta_Cliente + (Cantidad * Opc4);
        Dinero_Opc4 = Dinero_Opc4 + (Cantidad * Opc4);
        Cantidad_Chuletas = Cantidad_Chuletas – Cantidad;
        John_Collins = John_Collins – Cantidad;
        Mostrar << "Usted selecciono la Opción 4 (Chuletas de Cerdo a la BBQ y de bebida John Collins) con un precio de 25$";
      Caso (opcion = 5);
        Mostrar << "¿Cuántas ordenó?";
        Leer >> Cantidad;
        
        Mientras (Cantidad > Cantidad_Pasta OR Cantidad > Estracto_Picha) Hacer;
          Mostrar << "Solo hay ", Cantidad_Pasta, "Pastas Ex’ Reprobada y ", Estracto_Picha, "Botellas de Estracto de Picha disponibles";
          Leer >> Cantidad;
        Fin_Mientras
        
        Opc_Selec5 = Opc_Selec 5 + 1;
        Cuenta_Cliente = Cuenta_Cliente + (Cantidad * Opc5);
        Dinero_Opc5 = Dinero_Opc5 + (Cantidad * Opc5);
        Cantidad_Pasta = Cantidad_Pasta – Cantidad;
        Estracto_Picha = Estracto_Picha – Cantidad;
        Mostrar << "Usted selecciono la Opción 5 (Pasta Ex´ reprobada y de bebida Estracto de Picha) con un precio de 19$";
      Otro_Caso
        Mostrar << "¿Cuántas ordenó?";
        Leer >> Cantidad;
        
        Mientras (Cantidad > Cantidad_Guanábana OR Cantidad > Whisky_Maravilla) Hacer;
          Mostrar << "Solo hay ", Cantidad_Guanábana, "Ensaladas de Guanábana a la APK y ", Whisky_Maravilla, "Botellas de Whisky Maravilla disponibles";
          Leer >> Cantidad;
        Fin_Mientras
        
        Opc_Selec6 = Opc_Selec6 + 1;
        Cuenta_Cliente = Cuenta_Cliente + (Cantidad * Opc6);
        Dinero_Opc6 = Dinero_Opc6 + (Cantidad * Opc6);
        Cantidad_Guanábana = Cantidad_Guanábana – Cantidad;
        Whisky_Maravilla = Whisky_Maravilla – Cantidad;
        Mostrar << "Usted seleccionó la Opción 6 (Ensalada de Guanábana a la APK y de bebida Whisky Maravilla) con un precio de 27$";
    Fin_Caso
    
    Mostrar << "Desea añadir algo más a su menú?, 1: Sí, 0: No ";
    Leer >> R;
    Hasta (R = 0);
  Fin_Repetir
  
  Dinero_Todas_Opcs = Dinero_Opc1 + Dinero_Opc2 + Dinero_Opc3 + Dinero_Opc4 + Dinero_Opc5 + Dinero_Opc6;
  
  Mostrar << "Este es el dinero acumulado por cada una de las opciones";
  Mostrar << "Opcion 1:", Dinero_Opc1;
  Mostrar << "Opcion 2:", Dinero_Opc2;
  Mostrar << "Opcion 3:", Dinero_Opc3;
  Mostrar << "Opcion 4:", Dinero_Opc4;
  Mostrar << "Opcion 5:", Dinero_Opc5;
  Mostrar << "Opcion 6:", Dinero_Opc6;
  Mostrar << "Para un total de todas las opciones de: ", Dinero_Todas_Opcs;
Fin_Procedimiento
