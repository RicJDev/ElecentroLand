Procedimiento Eleccion(E/S: Modo_elegido);
Inicio
  Mostrar << "Bienvenido a la Ruleta que te hace pobre... (* sonido de toser *) Digo, La Ruleta MILLONARIA!!";
  Mostrar << "Ingrese la modalidad de juego seleccionando una de las siguientes, 1: Rojo y Negro, 2: Grupo de Números, 3: Numero a su eleccion";
  Leer >> Modo_elegido;
  
  Mientras (Modo_elegido < 1 OR Modo_elegido > 3) Hacer;
    Mostrar << "Modo de juego elegido incorrecto, Seleccione un modo de juego nuevamente";
    Leer >> Modo_elegido;
  Fin_Mientras
Fin_Procedimiento

Procedimiento Recoleccion(E: Modo_Elegido S: Apuesta, Seleccion);
Inicio
  En_Caso (Modo_Elegido > 0) Sea;
    Caso (Modo_Elegido = 1);
      Mostrar << "Ingrese una suma de fichas";
      Leer >> Apuesta;
      Mostrar << "Ingrese por cuanto quiere multiplicar su dinero (maximo 3) por cada vez que multiplique su dinero debera pagar un 150% extra de lo que aposto al perder";
      Leer >> Multiplicador;
      Mostrar << "Ingrese el color al cual apostara (1 : Rojo, 2 :Negro";
      Leer >> Seleccion;
      Mientras (Apuesta < 0) OR (Seleccion < 1 OR Seleccion > 2) Hacer;
        Mostrar << "Valores ingresados erroneos, ingreselos nuevamente";
        Leer >> Apuesta;
        Leer >> Seleccion;
      Fin_Mientras
    Caso (Modo_Elegido = 2);
      Mostrar << "Ingrese una suma de fichas";
      Leer >> Apuesta;
      Mostrar << "Ingrese por cuanto quiere multiplicar su dinero (maximo 3) por cada vez que multiplique su dinero debera pagar un 150% extra de lo que aposto al perder";
      Leer >> Multiplicador;
      Mostrar << "Ingrese el numero especifico al cual apostara (del 0 al 32)";
      Leer >> Seleccion;
      Mientras (Apuesta < 0) OR (Seleccion < 0 OR Seleccion > 32) Hacer;
        Mostrar << "Valores ingresados erroneos,ingreselos nuevamente";
        Leer >> Apuesta;
        Leer >> Seleccion;
      Fin_Mientras
    Otro_Caso
      Mostrar << "Ingrese una suma de fichas";
      Leer >> Apuesta;
      Mostrar << "Ingrese por cuanto quiere multiplicar su dinero (maximo 3) por cada vez que multiplique su dinero debera pagar un 150% extra de lo que aposto al perder";
      Leer >> Multiplicador;
      Mostrar << "Ingrese el grupo numero al cual apostarta (1: del 0 al 8, 2: del 9 al 16, 3: del 17 al 24, 4: del 25 al 32)";
      Leer >> Seleccion;
      Mientras (Apuesta < 0) OR (Seleccion < 1 OR Seleccion > 4) Hacer;
        Mostrar << "Valores ingresados erroneos, ;
        ingreselos nuevamente";
        Leer >> Apuesta;
        Leer >> Seleccion;
      Fin_Mientras
  Fin_Caso
Fin_Procedimiento
